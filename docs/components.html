Snippet

<input type="text" value="Hello World " id="myInput">
<button onclick="myFunction()">Copy</button>
<script>
function myFunction() {
var copyText = document.getElementById("myInput");
copyText.select();
document.execCommand("Copy");
alert("Copied the text: " + copyText.value);
}
</script>
